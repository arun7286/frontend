console.log(5 + 3); // addition
console.log(5 - 2); // Substraction
console.log(5 * 3); // multiplication
console.log(5 ** 3); // exponent
console.log(5 / 4);

// == (Loosely not Equality Operator)
console.log(10 == '10'); // true: Value + ignores datatype
console.log(10 != '10'); // false

// === (Strctly equal Operator)
console.log(10 === '10'); // false: Value + datatype.
console.log(10 !== '10'); //true

console.log(!true); // false
// Logical Operators.
console.log(10 || false); // 10
console.log(10 && false); // false

// Ternary Operator
// result = (Condition) ? Expression1 : Expression2;

result = (18 > 10) ? 'Expression 1 is true' : 'Expression 1 is false';
console.log(result);

//typeOf() operator Returns Datatype of an Object
var a = 100;
console.log(a, typeof a); // (100, number)

a = 100.45;
console.log(a, typeof a); // (100.45, number)

a = 'Kodnest';
console.log(a, typeof a);//(Kodnest, string)

a = 'k';
console.log(a, typeof a); //(Kodnest, String)

a = true;
console.log(a, typeof a); //(true, boolean)
