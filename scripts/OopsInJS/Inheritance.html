     class Employee {
        constructor(name,id) {
            this.name = name;
            this.id = id;
        }
        work() {
            console.log(this.name, "isworking")
        }
    }
    class EmployeePersonalInfo extends Employee {
        constructor(name, id,email,phone) {
            super(name, id);
            this.email = email;
            this.phone = phone;
        }
        getDetails() {
            console.log("Employee Name: " + this.name);
            console.log("Employee ID: " + this.id);
            console.log("Employee Email: " + this.email);
            console.log("Employee Phone: " + this.phone);
        }
    }
    let emp1 = new EmployeePersonalInfo("Arun", 25, "arun@gmail.com", 9948181771);
    emp1.getDetails();
