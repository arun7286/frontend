  /* In JavaScript, there are three main ways to declare a variable:
    1.var – an older way of declaring variables; it’s best avoided in modern JavaScript due to potential scope issues
          >The variable which are declared using var can be accessed inside the same block and outside the block as well.
   */
  {
    var product = 20;
    console.log(product);
  }
  product = 400;
  console.log(product);

  /*
    2. let – used to declare a variable that can be reassigned.
      > The variables declared using let can be accessed and can be re-assigned with in the same block.
   */
  {
    let product_count = 300;
    console.log("Product count is ",product_count, "Before increment");
    product_count++;
    console.log("Product Count is: ",product_count);
  }
  //console.log(product_count); Not allowed outside the block.
  /*
  3. const – used for variables whose values should not change.
   > The variables which are declared using const cannot be accessed outside the block
       And cannot be re-assigned within or outside the block.
  */
  {
    const name = "Amith";
    console.log("With in the block: ",name);
    // name = "Amith"; not allowed to re-assign for cons.
