    let arr = [10, 20, 30, 40];
    console.log(arr); //[10, 20, 30, 40,]
    console.log(Array.isArray(arr)); //true

    arr.push(50);
    console.log(arr); // [10, 20, 30, 40, 50]

    arr.pop(); // removes last element
    console.log(arr); //[10, 20, 30, 40]

    arr.shift(); // removes first element
    console.log(arr); //[20, 30, 40]

    arr.unshift(22); // adds element at first(0th index)
    console.log(arr); //[20, 30, 40]

    let nums = [1, 2, 3, 4, 5];
    console.log(nums);

    console.log("Deleted Elements: ",nums.splice(2, 3, 600, 400, 20)); // [3, 4, 5]
    // splice(start, deleteCount, Element to be added)
    console.log(nums); //[1, 2, 600, 400, 'Kodnest', 20]
