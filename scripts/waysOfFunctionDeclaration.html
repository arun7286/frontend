    // Function Declaration;
  function add1(a, b) {
    return a + b;
  }
  console.log("Addition From Function Declaration: ", add1(10, 20));
  // -------------------------------------------------------------------------------------
  // Function Expression:
  let add2 = function(a,b) {
    return a + b;
  }
  console.log("Addition From Function Expression: ", add2(10, 20));
  // -------------------------------------------------------------------------------------
  function mul(a,b) {
    return a * b;
  }
  console.log("Multiplication From Function declaration: ", mul(10, 20));
  // -------------------------------------------------------------------------------------
  let mul2 = function(a,b) {
    return a * b;
  }
  console.log("Multiplication From Function Expression: ", mul2(10, 20));
  // -------------------------------------------------------------------------------------
  // Arrow Function
  let ar = (a,b) => {
    console.log("Division From Arrow Function: ", a/b);
  }
  //calling
  ar(10, 20);
